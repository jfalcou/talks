  \begin{block}{Our goals}
   \begin{itemize}

\end{itemize}

  \begin{itemize}
  \footnotesize
  \item Leveraging what we learned in $NT^2$
  \item Demonstrated some impacts in term of performance
  \item Made using SIMD almost as simple than scalar
  \end{itemize}
